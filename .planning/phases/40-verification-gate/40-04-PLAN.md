    ```python
    """VERIFY-04 anisotropy audit: per-quartile MAP comparison.

    Reads the CRB CSV (for qS values), the per-event per-h diagnostic CSV (for
    combined likelihoods), and the overall h=0.73 posterior (for MAP_total and σ).
    Computes per-quartile MAP_q and flags any quartile with |MAP_q - MAP_total| > σ
    as a Stage-2 trigger (D-12 — NOT an abort).
    """
    from __future__ import annotations

    import json
    import math
    import sys
    from pathlib import Path

    import numpy as np
    import pandas as pd

    ROOT = Path(__file__).resolve().parents[2]
    sys.path.insert(0, str(ROOT))

    from master_thesis_code.bayesian_inference.evaluation_report import extract_baseline  # noqa: E402

    TS = (ROOT / ".planning" / "debug" / "verify_gate_timestamp.txt").read_text().strip()
    TRUE_H = 0.73

    # --- Inputs (CRB path from sys.argv[1] per W5 — no sed-i on committed Python) ---
    if len(sys.argv) < 2:
        sys.exit("usage: anisotropy_driver.py <CRB_CSV>")
    CRB_CSV = Path(sys.argv[1])
    DIAG_CSV = Path("simulations/diagnostics/event_likelihoods.csv")
    POSTERIORS_DIR = Path("simulations/posteriors")

    # --- Step A: Overall h=0.73 posterior gives MAP_total and σ ---
    snap = extract_baseline(posteriors_dir=POSTERIORS_DIR, crb_csv_path=None, true_h=TRUE_H)
    MAP_total = snap.map_h
    sigma = (snap.ci_upper - snap.ci_lower) / 2.0  # D-13.3: half of 68% CI width

    # --- Step B: Per-event qS from CRB CSV ---
    crb = pd.read_csv(CRB_CSV)
    # Column name tolerance: try 'qS', else lowercase variant
    col_qs = next((c for c in crb.columns if c.lower() == 'qs'), None)
    col_idx = next((c for c in crb.columns if c.lower() in ('detection_index', 'event_idx', 'event_index', 'index')), None)
    if col_qs is None or col_idx is None:
        sys.exit(f"ERROR: CRB CSV missing qS or detection_index column (found columns: {list(crb.columns)})")

    qS_series = crb[col_qs].to_numpy(dtype=float)
    event_idx_series = crb[col_idx].to_numpy(dtype=int)

    # |qS − π/2|
    dist_from_equator = np.abs(qS_series - math.pi / 2.0)

    # --- Step C: Equal-count quartile edges (D-14) ---
    q_edges = np.quantile(dist_from_equator, [0.00, 0.25, 0.50, 0.75, 1.00])
    quartile_labels = ["Q1 (nearest ecliptic equator)", "Q2", "Q3", "Q4 (furthest from equator)"]
    quartile_indices = [[] for _ in range(4)]
    for ev_idx, d in zip(event_idx_series, dist_from_equator):
        # rightmost=False on np.digitize → bin is [left, right); last bin we include rightmost
        q = np.searchsorted(q_edges[1:-1], d, side="right")
        q = min(q, 3)
        quartile_indices[q].append(int(ev_idx))

    # --- Step D: Per-event per-h likelihoods from diagnostic CSV ---
    diag = pd.read_csv(DIAG_CSV)
    # Fieldnames per Phase 38 SUMMARY line 47-49:
    # event_idx, h, f_i, L_cat_no_bh, L_cat_with_bh, L_comp, combined_no_bh, combined_with_bh
    required_cols = {"event_idx", "h", "combined_no_bh"}
    missing = required_cols - set(diag.columns)
    if missing:
        sys.exit(f"ERROR: diagnostic CSV missing columns: {missing}")

    # --- Step E: Per-quartile MAP via posterior re-combination ---
    # For each quartile, sum log(combined_no_bh) over member event_idx per h,
    # then argmax over h. (log-posterior = sum of log-likelihoods, assuming flat h prior.)
    h_grid = sorted(diag["h"].unique())

    per_quartile = []
    for q, (label, members) in enumerate(zip(quartile_labels, quartile_indices)):
        if len(members) == 0:
            per_quartile.append({
                "quartile": q + 1,
                "label": label,
                "edge_lower": float(q_edges[q]),
                "edge_upper": float(q_edges[q + 1]),
                "n_events": 0,
                "MAP_q": None,
                "delta_MAP_abs": None,
                "sigma": sigma,
                "stage_2_trigger": False,
                "note": "empty quartile",
            })
            continue

        subset = diag[diag["event_idx"].isin(members)].copy()
        # Sum log(combined_no_bh) over events per h
        # Guard against zeros/negatives: if combined_no_bh <= 0, treat as log = -inf → per-h sum = -inf
        subset["log_combined"] = np.where(
            subset["combined_no_bh"] > 0,
            np.log(subset["combined_no_bh"]),
            -np.inf,
        )
        log_posterior_q = subset.groupby("h")["log_combined"].sum()
        # Align to h_grid
        lp_array = np.array([log_posterior_q.get(h, -np.inf) for h in h_grid])
        MAP_q = float(h_grid[int(np.argmax(lp_array))])
        delta = abs(MAP_q - MAP_total)
        per_quartile.append({
            "quartile": q + 1,
            "label": label,
            "edge_lower": float(q_edges[q]),
            "edge_upper": float(q_edges[q + 1]),
            "n_events": len(members),
            "event_idx_members": members,
            "MAP_q": MAP_q,
            "delta_MAP_abs": float(delta),
            "sigma": float(sigma),
            "stage_2_trigger": bool(delta > sigma),
        })

    # --- Step F: Write artifacts ---
    DEBUG_DIR = ROOT / ".planning" / "debug"
    report_md = DEBUG_DIR / f"anisotropy_audit_{TS}.md"
    report_json = DEBUG_DIR / f"anisotropy_audit_{TS}.json"

    any_trigger = any(q["stage_2_trigger"] for q in per_quartile)
    verdict = "STAGE-2-TRIGGER" if any_trigger else "PASS"

    lines = [
        "# VERIFY-04: Anisotropy Audit",
        "",
        f"**Timestamp:** {TS}",
        f"**Phase:** 40 Wave 3",
        f"**Requirement:** VERIFY-04",
        f"**Rule (D-12):** `>1σ shift is a Stage-2 trigger for Phase 42 (not a blocker)`",
        "",
        "## Overall h=0.73 posterior",
        "",
        f"- `MAP_total` = {MAP_total:.4f}",
        f"- 68% CI: [{snap.ci_lower:.4f}, {snap.ci_upper:.4f}]",
        f"- σ = (CI_upper − CI_lower) / 2 = {sigma:.4f}",
        f"- N events = {snap.n_events}",
        "",
        "## Quartile edges (on |qS − π/2|, equal-count per D-14)",
        "",
        f"- Q1: [{q_edges[0]:.4f}, {q_edges[1]:.4f})",
        f"- Q2: [{q_edges[1]:.4f}, {q_edges[2]:.4f})",
        f"- Q3: [{q_edges[2]:.4f}, {q_edges[3]:.4f})",
        f"- Q4: [{q_edges[3]:.4f}, {q_edges[4]:.4f}]",
        "",
        "## Per-quartile MAP_q",
        "",
        "| # | Quartile label | N events | MAP_q | |MAP_q − MAP_total| | σ | Trigger (ΔMAP > σ)? |",
        "|---|----------------|----------|-------|---------------------|---|----------------------|",
    ]
    for q in per_quartile:
        map_q = "-" if q['MAP_q'] is None else f"{q['MAP_q']:.4f}"
        delta = "-" if q['delta_MAP_abs'] is None else f"{q['delta_MAP_abs']:.4f}"
        trig = "YES" if q['stage_2_trigger'] else "no"
        lines.append(
            f"| {q['quartile']} | {q['label']} | {q['n_events']} | {map_q} | {delta} | {sigma:.4f} | {trig} |"
        )
    lines += [
        "",
        "## Verdict",
        "",
        f"**VERIFY-04: {verdict}**",
        "",
    ]
    if any_trigger:
        lines += [
            "Per D-12, this is a Stage-2 trigger for **Phase 42** (Sky-Dependent Injection Campaign), NOT an abort condition. Phase 40 continues; Phase 41 can still run if its own VERIFY-05 trigger also fires.",
            "",
        ]
    else:
        lines += [
            "No quartile shows a >1σ MAP shift. Phase 42 is NOT triggered by anisotropy. If Phase 41 is also not triggered (VERIFY-05 check), both CAMP-* phases may be marked `skipped (not triggered)` per ROADMAP.md.",
            "",
        ]

    lines += [
        "## Links",
        "",
        f"- CRB CSV source: `{CRB_CSV}`",
        f"- Per-event per-h CSV: `{DIAG_CSV}`",
        f"- Combined h=0.73 posterior: `simulations/combined_posterior.json`",
        f"- Driver: `.planning/debug/anisotropy_driver_{TS}.py`",
        f"- Machine-readable: `.planning/debug/anisotropy_audit_{TS}.json`",
        "",
    ]

    report_md.write_text("\n".join(lines))
    report_json.write_text(json.dumps({
        "ts": TS,
        "true_h": TRUE_H,
        "MAP_total": MAP_total,
        "ci_lower": snap.ci_lower,
        "ci_upper": snap.ci_upper,
        "sigma": sigma,
        "n_events": snap.n_events,
        "quartile_edges": [float(e) for e in q_edges],
        "per_quartile": per_quartile,
        "any_stage_2_trigger": any_trigger,
        "verdict": verdict,
    }, indent=2))

    print(f"VERIFY-04 verdict: {verdict}")
    print(f"  sigma={sigma:.4f}; deltas = {[q['delta_MAP_abs'] for q in per_quartile]}")
    print(f"  report: {report_md}")
    ```
