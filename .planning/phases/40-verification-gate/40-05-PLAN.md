    ```python
    """VERIFY-05 quadrature-weight parser.

    Reads per-h log files from the Plan 40-03 sweep AND the Plan 40-02 h=0.73 pre-captured
    warnings file. Extracts STAT-04 WARNING lines via regex and writes a raw CSV with
    (h, event_idx, quadrature_weight_outside_grid_numerator, quadrature_weight_outside_grid_denominator, source_file).
    """
    from __future__ import annotations

    import csv
    import re
    import sys
    from pathlib import Path

    ROOT = Path(__file__).resolve().parents[2]
    TS = (ROOT / ".planning" / "debug" / "verify_gate_timestamp.txt").read_text().strip()

    # Regex matches STAT-04 WARNING line format from bayesian_statistics.py:1269-1275
    # Example matched line (note the em-dash may be rendered as "--" in some locales):
    # "2026-04-08 16:09:11,123 - bayesian_statistics - WARNING - Event 12: >5% quadrature weight outside P_det grid — numerator=0.234, denominator=0.567"
    WARN_RE = re.compile(
        r"Event (?P<ev>\d+): >5%[^0-9]+quadrature weight outside P_det grid[^0-9]+"
        r"numerator=(?P<num>[\d.]+)[^0-9]+denominator=(?P<den>[\d.]+)"
    )

    # h-value extraction from log filename: master_thesis_code_<TS>_h_0_<SUFFIX>.log
    H_RE = re.compile(r"master_thesis_code_\d+_\d+_h_(0_\d+)\.log$")

    def extract_h_from_filename(path: Path) -> float | None:
        m = H_RE.search(path.name)
        if not m:
            return None
        # Convert "0_735" → 0.735
        suffix = m.group(1)  # "0_735"
        numeric = suffix.replace("_", ".", 1)  # "0.735"
        try:
            return float(numeric)
        except ValueError:
            return None

    raw_rows: list[dict[str, object]] = []

    # --- Source 1: Plan 40-03 sweep logs ---
    sweep_logs = sorted((ROOT / "simulations").glob("master_thesis_code_*_h_0_*.log"))
    for lp in sweep_logs:
        h = extract_h_from_filename(lp)
        if h is None:
            print(f"WARN: could not extract h from {lp.name}", file=sys.stderr)
            continue
        with lp.open("r", errors="replace") as f:
            for line in f:
                m = WARN_RE.search(line)
                if not m:
                    continue
                raw_rows.append({
                    "h": h,
                    "event_idx": int(m.group("ev")),
                    "quadrature_weight_outside_grid_numerator": float(m.group("num")),
                    "quadrature_weight_outside_grid_denominator": float(m.group("den")),
                    "source_file": str(lp.relative_to(ROOT)),
                })

    # --- Source 2: Plan 40-02 pre-captured h=0.73 warnings file ---
    # This file is the grep output (each line already a WARNING); h is implicitly 0.73 since
    # the eval was called with --h_value 0.73. But the WARN_RE matches regardless.
    pre_captured = ROOT / ".planning" / "debug" / f"verify02_quadrature_warnings_{TS}.log"
    if pre_captured.exists():
        with pre_captured.open("r", errors="replace") as f:
            for line in f:
                m = WARN_RE.search(line)
                if not m:
                    continue
                # Dedup against sweep logs: if event_idx at h=0.73 already in raw_rows from the
                # master_thesis_code_*_h_0_73.log sweep log, skip. (The 40-02 eval wrote its own
                # log file too — so we'd double-count otherwise.)
                ev = int(m.group("ev"))
                already = any(
                    r["h"] == 0.73 and r["event_idx"] == ev
                    for r in raw_rows
                )
                if already:
                    continue
                raw_rows.append({
                    "h": 0.73,
                    "event_idx": ev,
                    "quadrature_weight_outside_grid_numerator": float(m.group("num")),
                    "quadrature_weight_outside_grid_denominator": float(m.group("den")),
                    "source_file": str(pre_captured.relative_to(ROOT)),
                })

    # --- Write raw CSV ---
    out = ROOT / ".planning" / "debug" / f"pdet_quadrature_raw_{TS}.csv"
    out.parent.mkdir(parents=True, exist_ok=True)
    with out.open("w", newline="") as f:
        fieldnames = [
            "h",
            "event_idx",
            "quadrature_weight_outside_grid_numerator",
            "quadrature_weight_outside_grid_denominator",
            "source_file",
        ]
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        writer.writerows(raw_rows)

    print(f"Parsed {len(raw_rows)} WARNING rows across {len(sweep_logs)} sweep logs")
    print(f"Raw CSV: {out}")
    if not raw_rows:
        print("INFO: zero STAT-04 warnings matched — either no event crossed 5% at any h, OR the")
        print("      WARN_RE regex needs an update. Inspect the per-h log files manually.")
    ```

    **Step 2 — Write the aggregator driver** `.planning/debug/pdet_quadrature_aggregator_${TS}.py`:
    ```python
    """VERIFY-05 aggregator: per-h stats, all-h stats, histogram, Phase 41 verdict."""
    from __future__ import annotations

    import json
    import math
    import sys
    from pathlib import Path

    import numpy as np
    import pandas as pd

    ROOT = Path(__file__).resolve().parents[2]
    TS = (ROOT / ".planning" / "debug" / "verify_gate_timestamp.txt").read_text().strip()
    TRUE_H = 0.73
    PHASE_41_THRESHOLD = 0.05

    # N_events_total passed in as argv[1] by the executor
    N_EVENTS_TOTAL = int(sys.argv[1])

    raw_csv = ROOT / ".planning" / "debug" / f"pdet_quadrature_raw_{TS}.csv"
    df = pd.read_csv(raw_csv) if raw_csv.exists() and raw_csv.stat().st_size > 0 else pd.DataFrame(
        columns=["h", "event_idx", "quadrature_weight_outside_grid_numerator",
                 "quadrature_weight_outside_grid_denominator", "source_file"]
    )

    # Per D-18, the Phase 41 trigger is on the NUMERATOR mean at h=0.73.
    # Unlogged events have weight <= 0.05; we treat unlogged as 0.0 (conservative lower bound).
    # Report both interpretations in the summary.

    # --- Per-h stats ---
    per_h_rows = []
    unique_h = sorted(df["h"].unique()) if not df.empty else []
    for h in unique_h:
        sub = df[df["h"] == h]
        # Logged-only stats (biased high):
        n_logged = len(sub)
        logged_mean_num = float(sub["quadrature_weight_outside_grid_numerator"].mean())
        logged_max_num = float(sub["quadrature_weight_outside_grid_numerator"].max())
        logged_mean_den = float(sub["quadrature_weight_outside_grid_denominator"].mean())
        logged_max_den = float(sub["quadrature_weight_outside_grid_denominator"].max())
        # All-events lower-bound stats (treat unlogged as 0):
        all_mean_num = float(sub["quadrature_weight_outside_grid_numerator"].sum() / N_EVENTS_TOTAL)
        all_mean_den = float(sub["quadrature_weight_outside_grid_denominator"].sum() / N_EVENTS_TOTAL)
        per_h_rows.append({
            "h": float(h),
            "n_events_above_5pct_numerator": int(sub["quadrature_weight_outside_grid_numerator"].gt(0.05).sum()),
            "n_events_above_5pct_denominator": int(sub["quadrature_weight_outside_grid_denominator"].gt(0.05).sum()),
            "n_events_logged": n_logged,
            "n_events_total": N_EVENTS_TOTAL,
            "logged_mean_numerator": logged_mean_num,
            "logged_max_numerator": logged_max_num,
            "logged_mean_denominator": logged_mean_den,
            "logged_max_denominator": logged_max_den,
            "all_events_mean_numerator_lb": all_mean_num,  # lower bound
            "all_events_mean_denominator_lb": all_mean_den,  # lower bound
        })

    # --- All-h aggregate ---
    if not df.empty:
        all_mean_num = float(df["quadrature_weight_outside_grid_numerator"].mean())
        all_max_num = float(df["quadrature_weight_outside_grid_numerator"].max())
        all_mean_den = float(df["quadrature_weight_outside_grid_denominator"].mean())
        all_max_den = float(df["quadrature_weight_outside_grid_denominator"].max())
    else:
        all_mean_num = all_max_num = all_mean_den = all_max_den = 0.0

    # --- Per-event histogram (aggregated across all h) ---
    hist_bins = np.linspace(0.0, 1.0, 11)  # 10 bins on [0, 1] per D-17
    if not df.empty:
        hist_counts_num, _ = np.histogram(df["quadrature_weight_outside_grid_numerator"], bins=hist_bins)
        hist_counts_den, _ = np.histogram(df["quadrature_weight_outside_grid_denominator"], bins=hist_bins)
    else:
        hist_counts_num = np.zeros(10, dtype=int)
        hist_counts_den = np.zeros(10, dtype=int)

    # --- Phase 41 trigger decision (D-18) ---
    # 3-way verdict (W1):
    #   NOT-TRIGGERED              if mean_073_num < 0.03
    #   PHASE-41-TRIGGER-BORDERLINE if 0.03 <= mean_073_num <= 0.05
    #   PHASE-41-TRIGGERED         if mean_073_num > 0.05
    # BORDERLINE means user must decide; 40-06 treats it as "pause, halt chain".
    BORDERLINE_LOW = 0.03
    BORDERLINE_HIGH = PHASE_41_THRESHOLD  # 0.05
    h_073_row = next((r for r in per_h_rows if abs(r["h"] - TRUE_H) < 1e-6), None)
    if h_073_row is None:
        mean_073_num = 0.0
        trigger_verdict = "NOT-TRIGGERED"
        trigger_rationale = "No h=0.73 data in CSV (no event >5% at h=0.73) — Phase 41 NOT triggered"
    else:
        mean_073_num = h_073_row["all_events_mean_numerator_lb"]
        if mean_073_num > BORDERLINE_HIGH:
            trigger_verdict = "PHASE-41-TRIGGERED"
            trigger_rationale = (
                f"mean_{{h=0.73}}(numerator) lower bound = {mean_073_num:.4f} > {BORDERLINE_HIGH:.2f}"
            )
        elif mean_073_num >= BORDERLINE_LOW:
            trigger_verdict = "PHASE-41-TRIGGER-BORDERLINE"
            trigger_rationale = (
                f"mean_{{h=0.73}}(numerator) lower bound = {mean_073_num:.4f} is inside borderline band "
                f"[{BORDERLINE_LOW:.2f}, {BORDERLINE_HIGH:.2f}] — user decides"
            )
        else:
            trigger_verdict = "NOT-TRIGGERED"
            trigger_rationale = (
                f"mean_{{h=0.73}}(numerator) lower bound = {mean_073_num:.4f} < {BORDERLINE_LOW:.2f}"
            )
    # Back-compat boolean for any consumer that still reads a bool: True iff strictly triggered.
    trigger = trigger_verdict == "PHASE-41-TRIGGERED"

    # --- Dominant events at h=0.73 ---
    dom_events = []
    if not df.empty:
        h73 = df[abs(df["h"] - TRUE_H) < 1e-6].sort_values("quadrature_weight_outside_grid_numerator", ascending=False)
        for _, row in h73.head(10).iterrows():
            dom_events.append({
                "event_idx": int(row["event_idx"]),
    ```
