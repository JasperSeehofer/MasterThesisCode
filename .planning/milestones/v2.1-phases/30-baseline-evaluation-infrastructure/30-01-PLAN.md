3. **`compute_credible_interval(h_values: list[float], log_posteriors: list[float], level: float = 0.68) -> tuple[float, float]`**: Convert log-posteriors to normalized probability (subtract max, exp, normalize via trapezoid). Compute CDF via cumulative trapezoid (`scipy.integrate.cumulative_trapezoid`). Find lower bound where CDF crosses `(1-level)/2` and upper bound where CDF crosses `(1+level)/2` using linear interpolation. Return (lower, upper).

4. **`extract_baseline(posteriors_dir: Path, crb_csv_path: Path | None = None, true_h: float = 0.73) -> BaselineSnapshot`**: Load posteriors, validate >= 3 h-values (per D-02), compute MAP h (peak of log-posterior), 68% CI, bias %, event count. If `crb_csv_path` provided, load the CRB CSV with pandas and extract per-event summaries: `{d_L, SNR, sigma_d_L_over_d_L, condition_number, quality_pass}` for each detection. Return `BaselineSnapshot`.

   Return path to markdown report.

   ```python
   def _save_baseline(working_directory: str) -> None:
       """Extract baseline metrics from existing posteriors and save as baseline.json."""
       from pathlib import Path
       from master_thesis_code.bayesian_inference.evaluation_report import extract_baseline

       posteriors_dir = Path(working_directory) / "simulations" / "posteriors"
       crb_csv = Path(working_directory) / "simulations" / "prepared_cramer_rao_bounds.csv"

       baseline = extract_baseline(
           posteriors_dir=posteriors_dir,
           crb_csv_path=crb_csv if crb_csv.exists() else None,
       )

       # D-06: store in .planning/debug/ for cross-phase reference
       import json
       project_root = Path(__file__).resolve().parents[1]
       debug_dir = project_root / ".planning" / "debug"
       debug_dir.mkdir(parents=True, exist_ok=True)
       output_path = debug_dir / "baseline.json"
       output_path.write_text(json.dumps(baseline.to_json(), indent=2))
       _ROOT_LOGGER.info("Baseline saved to %s (MAP h=%.4f, bias=%.1f%%)", output_path, baseline.map_h, baseline.bias_percent)
   ```

   ```python
   def _compare_baseline(working_directory: str, baseline_path: str) -> None:
       """Generate comparison report between baseline and current posteriors."""
       from pathlib import Path
       import json
       from master_thesis_code.bayesian_inference.evaluation_report import (
           BaselineSnapshot, extract_baseline, generate_comparison_report,
       )

       baseline_data = json.loads(Path(baseline_path).read_text())
       baseline = BaselineSnapshot.from_json(baseline_data)

       posteriors_dir = Path(working_directory) / "simulations" / "posteriors"
       crb_csv = Path(working_directory) / "simulations" / "prepared_cramer_rao_bounds.csv"

       current = extract_baseline(
           posteriors_dir=posteriors_dir,
           crb_csv_path=crb_csv if crb_csv.exists() else None,
       )

       # D-06: comparison reports also go to .planning/debug/
       project_root = Path(__file__).resolve().parents[1]
       output_dir = project_root / ".planning" / "debug"
       output_dir.mkdir(parents=True, exist_ok=True)
       report_path = generate_comparison_report(baseline, current, output_dir)
       _ROOT_LOGGER.info("Comparison report written to %s", report_path)

       # Also print summary to stdout
       print(f"\n{'='*60}")
       print(f"  Baseline: MAP h={baseline.map_h:.4f}, bias={baseline.bias_percent:+.1f}%")
       print(f"  Current:  MAP h={current.map_h:.4f}, bias={current.bias_percent:+.1f}%")
       print(f"  Delta:    MAP h={current.map_h - baseline.map_h:+.4f}, bias={current.bias_percent - baseline.bias_percent:+.1f}%")
       print(f"{'='*60}\n")
   ```
