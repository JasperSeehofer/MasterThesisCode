    ```python
    import json, glob, numpy as np
    files = sorted(glob.glob('simulations/posteriors/h_*.json'))
    results = []
    for f in files:
        data = json.loads(open(f).read())
        h = data['h']
        log_sum = 0.0
        for k, v in data.items():
            if k != 'h' and isinstance(v, list) and v and v[0] > 0:
                log_sum += np.log(v[0])
        results.append((h, log_sum))
    results.sort()
    map_h = max(results, key=lambda x: x[1])[0]
    print(f"MAP h (from posteriors/) = {map_h}")
    for h, lp in results:
        print(f"  h={h:.3f}  log_L={lp:.2f}")
    ```
