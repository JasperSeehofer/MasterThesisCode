    If 43-01-SUMMARY records branch_decision=BRANCH-B, run the following migration script.
    First back up the original: `cp simulations/prepared_cramer_rao_bounds.csv simulations/prepared_cramer_rao_bounds.csv.bak_equatorial`

    ```python
    import pandas as pd
    import numpy as np
    from astropy.coordinates import SkyCoord, BarycentricTrueEcliptic
    import astropy.units as u

    # --- prepared_cramer_rao_bounds.csv ---
    df = pd.read_csv('simulations/prepared_cramer_rao_bounds.csv')
    assert '_coord_frame' not in df.columns, "Already migrated — aborting to prevent double transform"

    # qS is equatorial colatitude = pi/2 - Dec; convert to Dec in degrees
    ra_rad = df['phiS'].values
    dec_rad = np.pi / 2.0 - df['qS'].values  # equatorial colatitude -> latitude

    # Coordinate transform: ICRS -> BarycentricTrueEcliptic J2000
    # This matches handler.py lines 574-648 (Phase 36 fix)
    # Ref: master_thesis_code/galaxy_catalogue/handler.py
    coord = SkyCoord(
        ra=np.degrees(ra_rad) * u.deg,
        dec=np.degrees(dec_rad) * u.deg,
        frame='icrs'
    )
    ecl = coord.transform_to(BarycentricTrueEcliptic(equinox='J2000'))

    # ecl.lon is ecliptic longitude (phiS equivalent), ecl.lat is ecliptic latitude
    # Convert back to radians; qS = pi/2 - lat (colatitude convention)
    df['phiS'] = np.radians(ecl.lon.deg) % (2.0 * np.pi)
    df['qS'] = np.pi / 2.0 - np.radians(ecl.lat.deg)
    df['_coord_frame'] = 'ecliptic_BarycentricTrue_J2000'

    # Validate ranges before writing
    assert df['qS'].between(0, np.pi).all(), f"qS out of [0,pi]: {df['qS'].describe()}"
    assert df['phiS'].between(0, 2*np.pi).all(), f"phiS out of [0,2*pi]: {df['phiS'].describe()}"

    df.to_csv('simulations/prepared_cramer_rao_bounds.csv', index=False)
    print(f"Migrated {len(df)} rows. qS: [{df['qS'].min():.4f}, {df['qS'].max():.4f}] rad. phiS: [{df['phiS'].min():.4f}, {df['phiS'].max():.4f}] rad.")
    ```
